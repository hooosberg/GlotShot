### 2.7 `src/services/LicenseManager.ts` — 核心重构

**当前**：完全基于 `sessionStorage` 的 mock 实现，每次重启失效。

**目标**：通过 Electron IPC 调用主进程 StoreKit API，持久化购买状态。

#### 完整替换后的代码：

```typescript
import { isFreeModel } from '../engines/webllmModels'

export type PaidFeature = 'model' | 'export' | 'polish' | 'autocomplete'
export type ExportFormat = 'pdf' | 'docx' | 'png' | 'jpg' | 'txt' | 'md'
export interface LicenseState {
    isPro: boolean
}

export interface FeatureAccessResult {
    allowed: boolean
    requiresPro: boolean
    reason?: string
}

export interface PolishQuotaInfo {
    used: number
    limit: number
    remaining: number
    resetAt: string
    isPro: boolean
}

const LICENSE_CHANGE_EVENT = 'witnote-license-change'

// 内存缓存，避免每次都 IPC 查询
let cachedProStatus: boolean = false

function getTomorrowStartISO(): string {
    const now = new Date()
    const tomorrow = new Date(now.getFullYear(), now.getMonth(), now.getDate() + 1, 0, 0, 0)
    return tomorrow.toISOString()
}

function emitLicenseChange(state: LicenseState): void {
    if (typeof window === 'undefined') return
    window.dispatchEvent(new CustomEvent<LicenseState>(LICENSE_CHANGE_EVENT, {
        detail: state
    }))
}

export class LicenseManager {
    /**
     * 应用启动时调用，从主进程同步购买状态
     */
    static async initialize(): Promise<void> {
        if (typeof window !== 'undefined' && window.api?.checkLicenseStatus) {
            cachedProStatus = await window.api.checkLicenseStatus()
            emitLicenseChange({ isPro: cachedProStatus })
        }
    }

    static isPro(): boolean {
        return cachedProStatus
    }

    static getState(): LicenseState {
        return { isPro: this.isPro() }
    }

    static onChange(listener: (state: LicenseState) => void): () => void {
        if (typeof window === 'undefined') {
            return () => undefined
        }

        const handler = (event: Event) => {
            const customEvent = event as CustomEvent<LicenseState>
            listener(customEvent.detail || this.getState())
        }

        window.addEventListener(LICENSE_CHANGE_EVENT, handler as EventListener)
        return () => window.removeEventListener(LICENSE_CHANGE_EVENT, handler as EventListener)
    }

    /**
     * 发起 MAS 内购
     * @returns 是否购买成功
     */
    static async purchasePro(): Promise<boolean> {
        if (typeof window === 'undefined' || !window.api?.purchasePro) {
            console.warn('Purchase API not available')
            return false
        }

        try {
            const success = await window.api.purchasePro()
            if (success) {
                cachedProStatus = true
                emitLicenseChange({ isPro: true })
            }
            return success
        } catch (err) {
            console.error('Purchase failed:', err)
            return false
        }
    }

    /**
     * 恢复购买（MAS 审核要求必须提供此功能）
     * @returns 是否恢复成功
     */
    static async restorePurchases(): Promise<boolean> {
        if (typeof window === 'undefined' || !window.api?.restorePurchases) {
            console.warn('Restore API not available')
            return false
        }

        try {
            const success = await window.api.restorePurchases()
            if (success) {
                cachedProStatus = true
                emitLicenseChange({ isPro: true })
            }
            return success
        } catch (err) {
            console.error('Restore failed:', err)
            return false
        }
    }

    // ========== 以下功能门控方法保持不变 ==========

    static checkModelAccess(modelId: string): FeatureAccessResult {
        if (isFreeModel(modelId) || this.isPro()) {
            return { allowed: true, requiresPro: false }
        }
        return {
            allowed: false,
            requiresPro: true,
            reason: 'This model requires Pro'
        }
    }

    static checkExportAccess(format: ExportFormat): FeatureAccessResult {
        if (format === 'md' || format === 'txt') {
            return { allowed: true, requiresPro: false }
        }
        if (this.isPro()) {
            return { allowed: true, requiresPro: false }
        }
        return {
            allowed: false,
            requiresPro: true,
            reason: 'This export format requires Pro'
        }
    }

    static checkFeatureAccess(feature: PaidFeature, data?: { modelId?: string; format?: ExportFormat }): FeatureAccessResult {
        if (feature === 'model') {
            if (!data?.modelId) return { allowed: false, requiresPro: true, reason: 'Missing model id' }
            return this.checkModelAccess(data.modelId)
        }
        if (feature === 'export') {
            if (!data?.format) return { allowed: false, requiresPro: true, reason: 'Missing export format' }
            return this.checkExportAccess(data.format)
        }
        if (feature === 'polish') {
            if (this.isPro()) return { allowed: true, requiresPro: false }
            return { allowed: false, requiresPro: true, reason: 'Writing polish requires Pro' }
        }
        if (feature === 'autocomplete') {
            if (this.isPro()) return { allowed: true, requiresPro: false }
            return { allowed: false, requiresPro: true, reason: 'Smart continuation requires Pro' }
        }
        return { allowed: false, requiresPro: true, reason: 'Unknown feature' }
    }

    static getPolishQuota(): PolishQuotaInfo {
        if (this.isPro()) {
            return {
                used: 0,
                limit: Number.MAX_SAFE_INTEGER,
                remaining: Number.MAX_SAFE_INTEGER,
                resetAt: getTomorrowStartISO(),
                isPro: true
            }
        }
        return {
            used: 0,
            limit: 0,
            remaining: 0,
            resetAt: getTomorrowStartISO(),
            isPro: false
        }
    }

    static consumePolishQuota(): FeatureAccessResult {
        if (this.isPro()) {
            return { allowed: true, requiresPro: false }
        }
        return {
            allowed: false,
            requiresPro: true,
            reason: 'Writing polish requires Pro'
        }
    }
}

export default LicenseManager
```

### 2.8 `electron/main.ts` — 新增 StoreKit IPC Handlers

在文件中现有的 IPC handler 区域添加以下代码：

```typescript
import { inAppPurchase } from 'electron'

// ========== MAS 内购相关 ==========

// 你在 App Store Connect 中创建的产品 ID
const MAS_PRODUCT_ID = 'com.zikedece.witnote.pro.lifetime'

// 购买状态持久化 key
const PURCHASE_STORE_KEY = 'witnote-pro-purchased'

/**
 * 检查授权状态
 */
ipcMain.handle('mas:check-license', async () => {
    // 优先检查本地持久化
    const stored = store.get(PURCHASE_STORE_KEY, false)
    if (stored) return true

    // 如果是 MAS 版本，验证 receipt
    if (isMAS) {
        try {
            // Electron inAppPurchase receipt 验证
            const receipt = inAppPurchase.getReceiptURL()
            if (receipt) {
                // receipt 存在说明有有效购买
                // 生产环境中可以进一步验证 receipt 内容
                return true
            }
        } catch {
            // receipt 验证失败
        }
    }

    return false
})

/**
 * 发起购买
 */
ipcMain.handle('mas:purchase', async () => {
    if (!isMAS) {
        // 非 MAS 版本不支持购买（DMG 版走其他支付渠道）
        return false
    }

    try {
        // 获取产品信息
        const products = await inAppPurchase.getProducts([MAS_PRODUCT_ID])
        if (!products || products.length === 0) {
            console.error('Product not found:', MAS_PRODUCT_ID)
            return false
        }

        // 发起购买
        return new Promise<boolean>((resolve) => {
            // 监听交易更新
            inAppPurchase.on('transactions-updated', (_event, transactions) => {
                for (const transaction of transactions) {
                    if (transaction.productIdentifier === MAS_PRODUCT_ID) {
                        switch (transaction.transactionState) {
                            case 'purchased':
                                // 购买成功，持久化状态
                                store.set(PURCHASE_STORE_KEY, true)
                                inAppPurchase.finishTransactionByDate(transaction.transactionDate)
                                resolve(true)
                                return
                            case 'failed':
                                resolve(false)
                                return
                            case 'deferred':
                                // 等待家长审批等情况
                                resolve(false)
                                return
                        }
                    }
                }
            })

            // 发起购买请求
            inAppPurchase.purchaseProduct(MAS_PRODUCT_ID)
        })
    } catch (err) {
        console.error('Purchase error:', err)
        return false
    }
})

/**
 * 恢复购买
 */
ipcMain.handle('mas:restore', async () => {
    if (!isMAS) return false

    try {
        return new Promise<boolean>((resolve) => {
            let found = false

            inAppPurchase.on('transactions-updated', (_event, transactions) => {
                for (const transaction of transactions) {
                    if (transaction.productIdentifier === MAS_PRODUCT_ID) {
                        if (transaction.transactionState === 'restored' ||
                            transaction.transactionState === 'purchased') {
                            found = true
                            store.set(PURCHASE_STORE_KEY, true)
                            inAppPurchase.finishTransactionByDate(transaction.transactionDate)
                        }
                    }
                }
            })

            inAppPurchase.restoreCompletedTransactions()

            // 给恢复操作一个超时
            setTimeout(() => {
                resolve(found)
            }, 10000)
        })
    } catch (err) {
        console.error('Restore error:', err)
        return false
    }
})
```
